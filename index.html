How to Deploy This as a Live Web App (Netlify)

Free, no account needed initially. You will need to create a free Netlify account to get a persistent URL. The whole process takes about 5 minutes.

Step 1 — Prepare your file

You already have the file: index.html (this file). Save it to a folder on your computer — name the folder something like epicurean-intake.

Step 2 — Go to Netlify Drop

Open your browser and go to: app.netlify.com/drop

No login required for the first deploy.

Step 3 — Drag and drop

Drag your entire epicurean-intake folder into the Netlify Drop zone. It will deploy instantly and give you a URL like https://random-name-12345.netlify.app

Step 4 — Create a free account to lock in the URL

Sign up for a free Netlify account to claim the site. You can then rename it to something memorable like epicurean-intake.netlify.app — free forever.

Step 5 — Updating the tool in future

Whenever I give you a new version of index.html, just drag and drop the folder again. The URL stays the same. Takes 30 seconds.

Auto-write to Google Sheets (optional upgrade)

To have this tool write rows directly to your Epicurean Star Award 2026 working file, you need a small Google Apps Script backend. Here is the code.gs you paste into Google Apps Script:

function writeRows(data) {
  const sheet = SpreadsheetApp
    .openById('YOUR_SHEET_ID_HERE')
    .getSheetByName('Nominations');

  // Find last row with a real nomination number in column A
  const colA = sheet.getRange('A:A').getValues();
  let lastDataRow = 1;
  let lastNomNo = 0;
  for (let i = 0; i < colA.length; i++) {
    const val = colA[i][0];
    if (val !== '' && !isNaN(val) && Number(val) > 0) {
      lastDataRow = i + 1;
      if (Number(val) > lastNomNo) lastNomNo = Number(val);
    }
  }

  data.rows.forEach((row, idx) => {
    const insertAt = lastDataRow + 1 + idx;
    const nomNo = lastNomNo + 1 + idx;
    sheet.insertRowBefore(insertAt);

    // Write columns A–L (1–12) — skip M (13) which has formula
    sheet.getRange(insertAt, 1, 1, 12).setValues([[
      nomNo,                                    // A: No.
      (row.brand||'').toUpperCase(),            // B: Company/Brand
      (row.category||'').toUpperCase(),         // C: Award Category
      row.seatsAllocated === 1 ? 1 : '',        // D: No. of seats allocated
      row.seats||'No',                          // E: Purchase additional seats
      '',                                       // F: RSVP?
      row.contact||'',                          // G: Contact Person
      row.designation||'',                      // H: Designation
      row.email||'',                            // I: Email
      row.phone||'',                            // J: Contact Number
      row.member||'',                           // K: RAS Member / Non-Member
      ''                                        // L: Hidden
    ]]);

    // Skip M (13) — Nomination Amount formula
    // Write column N (14) only
    sheet.getRange(insertAt, 14, 1, 1).setValues([[
      row.seats||'0'                            // N: Additional Seats/Table Amount
    ]]);

    // Skip O (15) GST formula, P (16) Total formula
    // Write columns Q–S (17–19)
    sheet.getRange(insertAt, 17, 1, 3).setValues([[
      row.billingName||'',                      // Q: Billing Name
      row.billingAddress||'',                   // R: Billing Address
      row.dc6||''                               // S: Dining Credit 6 pax
    ]]);

    // Skip T (20) — Dining Credit 2 pax formula
    // Skip U (21) — hidden
    // Leave V–X (22–24) blank — Logo, DC Sent, DC Received
  });
}

function doPost(e) {
  try {
    let data;
    if (e.postData && e.postData.contents) {
      data = JSON.parse(e.postData.contents);
    } else if (e.parameter && e.parameter.payload) {
      data = JSON.parse(e.parameter.payload);
    } else {
      throw new Error('No data received');
    }
    writeRows(data);
    return ContentService
      .createTextOutput(JSON.stringify({status:'ok',rows:data.rows.length}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService
      .createTextOutput(JSON.stringify({status:'error',error:err.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function doGet(e) {
  try {
    if (e.parameter && e.parameter.payload) {
      const data = JSON.parse(e.parameter.payload);
      writeRows(data);
      const callback = e.parameter.callback || '';
      const result = JSON.stringify({status:'ok',rows:data.rows.length});
      if (callback) {
        return ContentService
          .createTextOutput(callback + '(' + result + ')')
          .setMimeType(ContentService.MimeType.JAVASCRIPT);
      }
      return ContentService
        .createTextOutput(result)
        .setMimeType(ContentService.MimeType.JSON);
    }
  } catch(err) {
    const callback = e.parameter.callback || '';
    const result = JSON.stringify({status:'error',error:err.toString()});
    if (callback) {
      return ContentService
        .createTextOutput(callback + '(' + result + ')')
        .setMimeType(ContentService.MimeType.JAVASCRIPT);
    }
    return ContentService
      .createTextOutput(result)
      .setMimeType(ContentService.MimeType.JSON);
  }
  return ContentService
    .createTextOutput(JSON.stringify({status:'ok',message:'ESA 2026 Sheet Writer Active'}))
    .setMimeType(ContentService.MimeType.JSON);
}

To deploy code.gs and wire it to this tool:

  1. Open your Epicurean Star Award 2026 working file in Google Sheets
  2. Copy the Sheet ID from the URL — it's the long string between /d/ and /edit
  3. Go to Extensions → Apps Script → paste the code above → replace YOUR_SHEET_ID_HERE with your Sheet ID
  4. Click Deploy → New deployment → Web App → Execute as: Me → Who has access: Anyone → Deploy
  5. Copy the Web App URL (looks like https://script.google.com/macros/s/AKfy.../exec)
  6. Send the URL to James (or paste it into the SHEET_ENDPOINT constant at the top of the script in this file)

Once wired, clicking Generate will silently write the nomination rows directly to your working file. No more copy-pasting.
